#include <iostream>

using namespace std;


int main() {
	
            float rad;
            
            scanf ("%f",&rad);
            
            printf ("diameter="); 
            
            printf ("%f",2*rad);
            
            printf (" circumference=");
            
            printf ("%f",6.28*rad);
            
            printf (" area of circle=");
            
            printf ("%f",3.14*rad*rad);
	
            return 0;

}
